int main(){
  int max = 0;
  int min = 0;
  long long sum = 0;

  int i;
  int num;

  cin >> i;

  while(i > 0){
    cin >> num;

    if(min == 0 && max == 0 && sum == 0){
      min = num;
      max = num;
    }
    if(num > max){
      max = num;
    }
    if(num < min){
      min = num;
    }

    sum += num;
    i--;
  }



  cout << min << " " << max << " " << sum << endl;
}
